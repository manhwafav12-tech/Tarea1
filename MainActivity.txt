data class ArchivoRecibido(
    val nombre: String,
    val ruta: String,
    val mime: String
)

class MainActivity : ComponentActivity() {

    private var archivoRecibido = mutableStateOf<ArchivoRecibido?>(null)

    private val pedirPermisoNotificaciones =
        registerForActivityResult(ActivityResultContracts.RequestPermission()) { }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        crearCanalNotificaciones(this)
        pedirPermisoNotificacionesSiHaceFalta()

        archivoRecibido.value = copiarArchivoRecibido(this, intent)

        setContent {
            val tareaViewModel: TareaViewModel = viewModel()

            AppTareas(
                viewModel = tareaViewModel,
                archivoRecibido = archivoRecibido.value
            )
        }
    }

    override fun onNewIntent(intent: Intent) {
        super.onNewIntent(intent)
        setIntent(intent)

        archivoRecibido.value = copiarArchivoRecibido(this, intent)
    }

    private fun pedirPermisoNotificacionesSiHaceFalta() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
            if (checkSelfPermission(Manifest.permission.POST_NOTIFICATIONS)
                != PackageManager.PERMISSION_GRANTED
            ) {
                pedirPermisoNotificaciones.launch(Manifest.permission.POST_NOTIFICATIONS)
            }
        }
    }
}

@Composable
fun AppTareas(
    viewModel: TareaViewModel,
    archivoRecibido: ArchivoRecibido?
) {
    var temaOscuro by remember {
        mutableStateOf(viewModel.temaOscuroInicial)
    }

    var notificaciones by remember {
        mutableStateOf(viewModel.notificacionesInicial)
    }

    val lifecycleOwner = LocalLifecycleOwner.current

    DisposableEffect(lifecycleOwner, temaOscuro, notificaciones) {
        val observer = LifecycleEventObserver { _, event ->
            if (event == Lifecycle.Event.ON_STOP) {
                viewModel.guardarPreferencias(
                    temaOscuro = temaOscuro,
                    notificaciones = notificaciones
                )
            }
        }

        lifecycleOwner.lifecycle.addObserver(observer)

        onDispose {
            lifecycleOwner.lifecycle.removeObserver(observer)
        }
    }

    MaterialTheme(
        colorScheme = if (temaOscuro) darkColorScheme() else lightColorScheme()
    ) {
        Surface(
            modifier = Modifier.fillMaxSize(),
            color = if (temaOscuro) {
                MaterialTheme.colorScheme.background
            } else {
                Color(0xFFF6F7FB)
            }
        ) {
            PantallaTareas(
                viewModel = viewModel,
                archivoRecibido = archivoRecibido,
                temaOscuro = temaOscuro,
                onTemaOscuroChange = {
                    temaOscuro = it
                    viewModel.guardarPreferencias(temaOscuro, notificaciones)
                },
                notificaciones = notificaciones,
                onNotificacionesChange = {
                    notificaciones = it
                    viewModel.guardarPreferencias(temaOscuro, notificaciones)
                }
            )
        }
    }
}

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun PantallaTareas(
    viewModel: TareaViewModel,
    archivoRecibido: ArchivoRecibido?,
    temaOscuro: Boolean,
    onTemaOscuroChange: (Boolean) -> Unit,
    notificaciones: Boolean,
    onNotificacionesChange: (Boolean) -> Unit
) {
    val context = LocalContext.current
    val tareas by viewModel.tareas.collectAsState()

    val totalTareas = tareas.size
    val tareasPendientes = tareas.count { !it.estaCompletada }
    val tareasCompletadas = tareas.count { it.estaCompletada }

    var titulo by remember { mutableStateOf("") }
    var descripcion by remember { mutableStateOf("") }
    var fecha by remember { mutableStateOf("") }
    var busqueda by remember { mutableStateOf("") }

    var archivoNombre by remember { mutableStateOf<String?>(null) }
    var archivoRuta by remember { mutableStateOf<String?>(null) }
    var archivoMime by remember { mutableStateOf<String?>(null) }
    var mostrarArchivoRecibido by remember { mutableStateOf(false) }

    var mostrarCalendario by remember { mutableStateOf(false) }
    val datePickerState = rememberDatePickerState()

    var tareaSeleccionada by remember { mutableStateOf<Tarea?>(null) }
    var tareaEditando by remember { mutableStateOf<Tarea?>(null) }
    var tareaCompartir by remember { mutableStateOf<Tarea?>(null) }

    val tareasFiltradas = remember(tareas, busqueda) {
        val texto = busqueda.trim()

        if (texto.isBlank()) {
            tareas
        } else {
            tareas.filter { tarea ->
                tarea.id.toString().contains(texto, ignoreCase = true) ||
                        tarea.titulo.contains(texto, ignoreCase = true) ||
                        tarea.descripcion.contains(texto, ignoreCase = true) ||
                        tarea.fecha.contains(texto, ignoreCase = true) ||
                        (tarea.archivoNombre ?: "").contains(texto, ignoreCase = true)
            }
        }
    }

    LaunchedEffect(archivoRecibido) {
        if (archivoRecibido != null) {
            archivoNombre = archivoRecibido.nombre
            archivoRuta = archivoRecibido.ruta
            archivoMime = archivoRecibido.mime
            mostrarArchivoRecibido = true

            if (titulo.isBlank()) {
                titulo = archivoRecibido.nombre
            }

            if (descripcion.isBlank()) {
                descripcion = "Archivo adjunto: ${archivoRecibido.nombre}"
            }
        }
    }

    if (mostrarCalendario) {
        DatePickerDialog(
            onDismissRequest = {
                mostrarCalendario = false
            },
            confirmButton = {
                TextButton(
                    onClick = {
                        val millis = datePickerState.selectedDateMillis
                        if (millis != null) {
                            fecha = formatearFecha(millis)
                        }
                        mostrarCalendario = false
                    }
                ) {
                    Text("Aceptar")
                }
            },
            dismissButton = {
                TextButton(
                    onClick = {
                        mostrarCalendario = false
                    }
                ) {
                    Text("Cancelar")
                }
            }
        ) {
            DatePicker(state = datePickerState)
        }
    }

    tareaSeleccionada?.let { tarea ->
        AlertDialog(
            onDismissRequest = {
                tareaSeleccionada = null
            },
            title = {
                Text("Vista previa")
            },
            text = {
                Column {
                    Text(
                        text = tarea.titulo,
                        style = MaterialTheme.typography.titleLarge,
                        fontWeight = FontWeight.Bold,
                        textDecoration = if (tarea.estaCompletada) {
                            TextDecoration.LineThrough
                        } else {
                            TextDecoration.None
                        }
                    )

                    Spacer(modifier = Modifier.height(10.dp))

                    Text("Descripción", fontWeight = FontWeight.Bold)

                    Text(
                        text = tarea.descripcion.ifBlank { "Sin descripción" },
                        style = MaterialTheme.typography.bodyMedium
                    )

                    Spacer(modifier = Modifier.height(10.dp))

                    Text("Fecha: ${tarea.fecha.ifBlank { "Sin fecha" }}")

                    if (!tarea.archivoNombre.isNullOrBlank()) {
                        Spacer(modifier = Modifier.height(10.dp))

                        Text("Archivo", fontWeight = FontWeight.Bold)
                        Text(tarea.archivoNombre)
                    }

                    Spacer(modifier = Modifier.height(10.dp))

                    AssistChip(
                        onClick = {},
                        label = {
                            Text(
                                if (tarea.estaCompletada) {
                                    "Completada"
                                } else {
                                    "Pendiente"
                                }
                            )
                        }
                    )
                }
            },
            confirmButton = {
                TextButton(
                    onClick = {
                        tareaEditando = tarea
                        tareaSeleccionada = null
                    }
                ) {
                    Text("Editar")
                }
            },
            dismissButton = {
                TextButton(
                    onClick = {
                        tareaSeleccionada = null
                    }
                ) {
                    Text("Cerrar")
                }
            }
        )
    }

    tareaCompartir?.let { tarea ->
        AlertDialog(
            onDismissRequest = {
                tareaCompartir = null
            },
            title = {
                Text("Compartir tarea")
            },
            text = {
                Text("Elige el formato para compartir esta tarea.")
            },
            confirmButton = {
                TextButton(
                    onClick = {
                        compartirTareaPdf(context, tarea)
                        tareaCompartir = null
                    }
                ) {
                    Text("PDF")
                }
            },
            dismissButton = {
                TextButton(
                    onClick = {
                        compartirTareaTxt(context, tarea)
                        tareaCompartir = null
                    }
                ) {
                    Text("TXT")
                }
            }
        )
    }

    tareaEditando?.let { tarea ->
        var tituloEditado by remember(tarea.id) {
            mutableStateOf(tarea.titulo)
        }

        var descripcionEditada by remember(tarea.id) {
            mutableStateOf(tarea.descripcion)
        }

        var fechaEditada by remember(tarea.id) {
            mutableStateOf(tarea.fecha)
        }

        var archivoNombreEditado by remember(tarea.id) {
            mutableStateOf(tarea.archivoNombre ?: "")
        }

        var mostrarCalendarioEditar by remember {
            mutableStateOf(false)
        }

        val datePickerEditarState = rememberDatePickerState()

        if (mostrarCalendarioEditar) {
            DatePickerDialog(
                onDismissRequest = {
                    mostrarCalendarioEditar = false
                },
                confirmButton = {
                    TextButton(
                        onClick = {
                            val millis = datePickerEditarState.selectedDateMillis
                            if (millis != null) {
                                fechaEditada = formatearFecha(millis)
                            }
                            mostrarCalendarioEditar = false
                        }
                    ) {
                        Text("Aceptar")
                    }
                },
                dismissButton = {
                    TextButton(
                        onClick = {
                            mostrarCalendarioEditar = false
                        }
                    ) {
                        Text("Cancelar")
                    }
                }
            ) {
                DatePicker(state = datePickerEditarState)
            }
        }

        AlertDialog(
            onDismissRequest = {
                tareaEditando = null
            },
            title = {
                Text("Editar tarea")
            },
            text = {
                LazyColumn(
                    verticalArrangement = Arrangement.spacedBy(10.dp)
                ) {
                    item {
                        OutlinedTextField(
                            value = tituloEditado,
                            onValueChange = { tituloEditado = it },
                            label = { Text("Título") },
                            modifier = Modifier.fillMaxWidth(),
                            singleLine = true,
                            shape = RoundedCornerShape(14.dp)
                        )
                    }

                    item {
                        OutlinedTextField(
                            value = descripcionEditada,
                            onValueChange = { descripcionEditada = it },
                            label = { Text("Descripción") },
                            placeholder = { Text("Edita los detalles de la tarea...") },
                            modifier = Modifier
                                .fillMaxWidth()
                                .height(150.dp),
                            minLines = 5,
                            maxLines = 8,
                            shape = RoundedCornerShape(14.dp)
                        )
                    }

                    item {
                        Box(
                            modifier = Modifier.fillMaxWidth()
                        ) {
                            OutlinedTextField(
                                value = fechaEditada,
                                onValueChange = {},
                                label = { Text("Fecha") },
                                readOnly = true,
                                modifier = Modifier.fillMaxWidth(),
                                shape = RoundedCornerShape(14.dp)
                            )

                            Box(
                                modifier = Modifier
                                    .matchParentSize()
                                    .clickable {
                                        mostrarCalendarioEditar = true
                                    }
                            )
                        }
                    }

                    if (!tarea.archivoRuta.isNullOrBlank()) {
                        item {
                            OutlinedTextField(
                                value = archivoNombreEditado,
                                onValueChange = { archivoNombreEditado = it },
                                label = { Text("Nombre del archivo") },
                                modifier = Modifier.fillMaxWidth(),
                                singleLine = true,
                                shape = RoundedCornerShape(14.dp)
                            )
                        }
                    }
                }
            },
            confirmButton = {
                TextButton(
                    onClick = {
                        val tareaActualizada = tarea.copy(
                            titulo = tituloEditado,
                            descripcion = descripcionEditada,
                            fecha = fechaEditada,
                            archivoNombre = archivoNombreEditado.ifBlank {
                                tarea.archivoNombre
                            }
                        )

                        viewModel.editarTarea(tareaActualizada)
                        tareaEditando = null
                    }
                ) {
                    Text("Guardar")
                }
            },
            dismissButton = {
                TextButton(
                    onClick = {
                        tareaEditando = null
                    }
                ) {
                    Text("Cancelar")
                }
            }
        )
    }

    LazyColumn(
        modifier = Modifier.fillMaxSize(),
        contentPadding = PaddingValues(20.dp),
        verticalArrangement = Arrangement.spacedBy(16.dp)
    ) {
        item {
            Encabezado(
                temaOscuro = temaOscuro,
                onTemaOscuroChange = onTemaOscuroChange,
                notificaciones = notificaciones,
                onNotificacionesChange = onNotificacionesChange
            )
        }

        item {
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(10.dp)
            ) {
                TarjetaResumen(
                    titulo = "Total",
                    cantidad = totalTareas,
                    colorFondo = Color(0xFFECEFFF),
                    colorTexto = Color(0xFF5B5FF0),
                    modifier = Modifier.weight(1f)
                )

                TarjetaResumen(
                    titulo = "Pendientes",
                    cantidad = tareasPendientes,
                    colorFondo = Color(0xFFFFF3C7),
                    colorTexto = Color(0xFFF0A500),
                    modifier = Modifier.weight(1f)
                )

                TarjetaResumen(
                    titulo = "Listas",
                    cantidad = tareasCompletadas,
                    colorFondo = Color(0xFFD7F8E8),
                    colorTexto = Color(0xFF13B67A),
                    modifier = Modifier.weight(1f)
                )
            }
        }

        if (mostrarArchivoRecibido && archivoNombre != null) {
            item {
                Card(
                    modifier = Modifier.fillMaxWidth(),
                    shape = RoundedCornerShape(22.dp),
                    elevation = CardDefaults.cardElevation(defaultElevation = 5.dp),
                    colors = CardDefaults.cardColors(
                        containerColor = Color(0xFFEAF2FF)
                    )
                ) {
                    Column(
                        modifier = Modifier.padding(16.dp)
                    ) {
                        Text(
                            text = "📎 Archivo recibido",
                            style = MaterialTheme.typography.titleMedium,
                            fontWeight = FontWeight.Bold,
                            color = Color(0xFF2457C5)
                        )

                        Spacer(modifier = Modifier.height(8.dp))

                        Text(
                            text = archivoNombre ?: "Archivo",
                            style = MaterialTheme.typography.bodyMedium,
                            color = Color(0xFF333333)
                        )

                        Spacer(modifier = Modifier.height(10.dp))

                        Row(
                            horizontalArrangement = Arrangement.spacedBy(8.dp),
                            modifier = Modifier.fillMaxWidth()
                        ) {
                            FilledTonalButton(
                                onClick = {
                                    titulo = archivoNombre ?: titulo
                                    descripcion = "Archivo adjunto: ${archivoNombre ?: ""}"
                                },
                                modifier = Modifier.weight(1f),
                                shape = RoundedCornerShape(12.dp)
                            ) {
                                Text("Usar")
                            }

                            OutlinedButton(
                                onClick = {
                                    archivoNombre = null
                                    archivoRuta = null
                                    archivoMime = null
                                    mostrarArchivoRecibido = false
                                },
                                modifier = Modifier.weight(1f),
                                shape = RoundedCornerShape(12.dp)
                            ) {
                                Text("Quitar")
                            }
                        }
                    }
                }
            }
        }

        item {
            Card(
                modifier = Modifier.fillMaxWidth(),
                shape = RoundedCornerShape(24.dp),
                elevation = CardDefaults.cardElevation(defaultElevation = 6.dp),
                colors = CardDefaults.cardColors(
                    containerColor = MaterialTheme.colorScheme.surface
                )
            ) {
                Column(
                    modifier = Modifier.padding(18.dp)
                ) {
                    Row(
                        verticalAlignment = Alignment.CenterVertically
                    ) {
                        Card(
                            shape = RoundedCornerShape(50),
                            colors = CardDefaults.cardColors(
                                containerColor = Color(0xFF5B5FF0)
                            )
                        ) {
                            Text(
                                text = "+",
                                color = Color.White,
                                fontWeight = FontWeight.Bold,
                                modifier = Modifier.padding(
                                    horizontal = 15.dp,
                                    vertical = 8.dp
                                )
                            )
                        }

                        Spacer(modifier = Modifier.width(10.dp))

                        Column {
                            Text(
                                text = "Nueva tarea",
                                style = MaterialTheme.typography.titleLarge,
                                fontWeight = FontWeight.Bold
                            )

                            Text(
                                text = "Completa los datos de tu pendiente",
                                style = MaterialTheme.typography.bodySmall,
                                color = MaterialTheme.colorScheme.onSurface.copy(alpha = 0.6f)
                            )
                        }
                    }

                    Spacer(modifier = Modifier.height(16.dp))

                    OutlinedTextField(
                        value = titulo,
                        onValueChange = { titulo = it },
                        label = { Text("Título de la tarea") },
                        placeholder = { Text("Ej: Revisar documentación") },
                        modifier = Modifier.fillMaxWidth(),
                        singleLine = true,
                        shape = RoundedCornerShape(16.dp)
                    )

                    Spacer(modifier = Modifier.height(10.dp))

                    OutlinedTextField(
                        value = descripcion,
                        onValueChange = { descripcion = it },
                        label = { Text("Descripción") },
                        placeholder = { Text("Escribe los detalles de la tarea...") },
                        modifier = Modifier
                            .fillMaxWidth()
                            .height(150.dp),
                        minLines = 5,
                        maxLines = 8,
                        shape = RoundedCornerShape(16.dp)
                    )

                    Spacer(modifier = Modifier.height(10.dp))

                    Box(
                        modifier = Modifier.fillMaxWidth()
                    ) {
                        OutlinedTextField(
                            value = fecha,
                            onValueChange = {},
                            label = { Text("Fecha") },
                            placeholder = { Text("Seleccionar fecha") },
                            readOnly = true,
                            modifier = Modifier.fillMaxWidth(),
                            shape = RoundedCornerShape(16.dp)
                        )

                        Box(
                            modifier = Modifier
                                .matchParentSize()
                                .clickable {
                                    mostrarCalendario = true
                                }
                        )
                    }

                    if (!archivoNombre.isNullOrBlank()) {
                        Spacer(modifier = Modifier.height(10.dp))

                        Text(
                            text = "📎 Adjunto: $archivoNombre",
                            style = MaterialTheme.typography.bodySmall,
                            color = Color(0xFF2457C5),
                            fontWeight = FontWeight.Bold
                        )
                    }

                    Spacer(modifier = Modifier.height(16.dp))

                    Button(
                        onClick = {
                            if (titulo.isNotBlank()) {
                                viewModel.agregarTarea(
                                    titulo = titulo,
                                    descripcion = descripcion,
                                    fecha = fecha,
                                    archivoNombre = archivoNombre,
                                    archivoRuta = archivoRuta,
                                    archivoMime = archivoMime
                                )

                                if (notificaciones) {
                                    mostrarNotificacionTarea(
                                        context = context,
                                        titulo = "Tarea agregada",
                                        mensaje = "Se agregó: $titulo"
                                    )
                                }

                                titulo = ""
                                descripcion = ""
                                fecha = ""
                                archivoNombre = null
                                archivoRuta = null
                                archivoMime = null
                                mostrarArchivoRecibido = false
                            }
                        },
                        modifier = Modifier
                            .fillMaxWidth()
                            .height(54.dp),
                        shape = RoundedCornerShape(16.dp),
                        colors = ButtonDefaults.buttonColors(
                            containerColor = Color(0xFF5B5FF0)
                        )
                    ) {
                        Text(
                            text = "+ Agregar tarea",
                            fontWeight = FontWeight.Bold
                        )
                    }
                }
            }
        }

        item {
            OutlinedTextField(
                value = busqueda,
                onValueChange = { busqueda = it },
                label = { Text("Buscar tarea") },
                placeholder = { Text("Buscar por id, título, descripción, fecha o archivo") },
                modifier = Modifier.fillMaxWidth(),
                singleLine = true,
                shape = RoundedCornerShape(16.dp)
            )
        }

        item {
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                OutlinedButton(
                    onClick = {
                        compartirTareasTxt(context, tareas)
                    },
                    modifier = Modifier.weight(1f),
                    enabled = tareas.isNotEmpty(),
                    shape = RoundedCornerShape(12.dp)
                ) {
                    Text("Compartir TXT")
                }

                OutlinedButton(
                    onClick = {
                        compartirTareasPdf(context, tareas)
                    },
                    modifier = Modifier.weight(1f),
                    enabled = tareas.isNotEmpty(),

fun formatearFecha(millis: Long): String {
    val formato = SimpleDateFormat("dd/MM/yyyy", Locale.getDefault())
    return formato.format(Date(millis))
}

fun crearCanalNotificaciones(context: Context) {
    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
        val canal = NotificationChannel(
            CANAL_TAREAS_ID,
            "Notificaciones de tareas",
            NotificationManager.IMPORTANCE_DEFAULT
        ).apply {
            description = "Canal para avisos de tareas"
        }

        val notificationManager =
            context.getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager

        notificationManager.createNotificationChannel(canal)
    }
}

fun mostrarNotificacionTarea(
    context: Context,
    titulo: String,
    mensaje: String
) {
    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
        if (context.checkSelfPermission(Manifest.permission.POST_NOTIFICATIONS)
            != PackageManager.PERMISSION_GRANTED
        ) {
            return
        }
    }

    val notificacion = NotificationCompat.Builder(context, CANAL_TAREAS_ID)
        .setSmallIcon(android.R.drawable.ic_dialog_info)
        .setContentTitle(titulo)
        .setContentText(mensaje)
        .setPriority(NotificationCompat.PRIORITY_DEFAULT)
        .setAutoCancel(true)
        .build()

    NotificationManagerCompat.from(context).notify(
        System.currentTimeMillis().toInt(),
        notificacion
    )
}

fun copiarArchivoRecibido(
    context: Context,
    intent: Intent?
): ArchivoRecibido? {
    if (intent == null) return null
    if (intent.action != Intent.ACTION_SEND) return null

    val uri = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
        intent.getParcelableExtra(Intent.EXTRA_STREAM, Uri::class.java)
    } else {
        @Suppress("DEPRECATION")
        intent.getParcelableExtra(Intent.EXTRA_STREAM)
    } ?: return null

    val nombreOriginal = obtenerNombreArchivo(context, uri)
    val mime = intent.type ?: context.contentResolver.getType(uri) ?: "application/octet-stream"

    val carpeta = File(context.filesDir, "archivos_tareas")
    if (!carpeta.exists()) carpeta.mkdirs()

    val archivoDestino = File(
        carpeta,
        "${System.currentTimeMillis()}_$nombreOriginal"
    )

    context.contentResolver.openInputStream(uri)?.use { input ->
        FileOutputStream(archivoDestino).use { output ->
            input.copyTo(output)
        }
    }

    return ArchivoRecibido(
        nombre = nombreOriginal,
        ruta = archivoDestino.absolutePath,
        mime = mime
    )
}

fun obtenerNombreArchivo(context: Context, uri: Uri): String {
    var nombre = "archivo_recibido"

    val cursor = context.contentResolver.query(
        uri,
        null,
        null,
        null,
        null
    )

    cursor?.use {
        val indiceNombre = it.getColumnIndex(OpenableColumns.DISPLAY_NAME)

        if (indiceNombre >= 0 && it.moveToFirst()) {
            nombre = it.getString(indiceNombre)
        }
    }

    return nombre
}

fun abrirArchivo(context: Context, tarea: Tarea) {
    val ruta = tarea.archivoRuta ?: return
    val archivo = File(ruta)

    if (!archivo.exists()) return

    val uri = FileProvider.getUriForFile(
        context,
        "${context.packageName}.fileprovider",
        archivo
    )

    val intent = Intent(Intent.ACTION_VIEW).apply {
        setDataAndType(uri, tarea.archivoMime ?: "*/*")
        addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION)
    }

    context.startActivity(
        Intent.createChooser(intent, "Abrir archivo")
    )
}

fun compartirTareasPdf(context: Context, tareas: List<Tarea>) {
    val pdfDocument = PdfDocument()

    val textoNormal = Paint().apply {
        textSize = 14f
    }

    val textoTitulo = Paint().apply {
        textSize = 22f
        isFakeBoldText = true
    }

    val textoSubtitulo = Paint().apply {
        textSize = 16f
        isFakeBoldText = true
    }

    var numeroPagina = 1
    var pageInfo = PdfDocument.PageInfo.Builder(595, 842, numeroPagina).create()
    var page = pdfDocument.startPage(pageInfo)
    var canvas = page.canvas

    var y = 50

    canvas.drawText("Lista de Tareas", 40f, y.toFloat(), textoTitulo)
    y += 40

    if (tareas.isEmpty()) {
        canvas.drawText("No hay tareas registradas.", 40f, y.toFloat(), textoNormal)
    } else {
        tareas.forEachIndexed { index, tarea ->
            if (y > 760) {
                pdfDocument.finishPage(page)

                numeroPagina++
                pageInfo = PdfDocument.PageInfo.Builder(595, 842, numeroPagina).create()
                page = pdfDocument.startPage(pageInfo)
                canvas = page.canvas
                y = 50
            }

            canvas.drawText("${index + 1}. ${tarea.titulo}", 40f, y.toFloat(), textoSubtitulo)
            y += 24

            canvas.drawText(
                "Descripción: ${tarea.descripcion.ifBlank { "Sin descripción" }}",
                40f,
                y.toFloat(),
                textoNormal
            )
            y += 22

            canvas.drawText(
                "Fecha: ${tarea.fecha.ifBlank { "Sin fecha" }}",
                40f,
                y.toFloat(),
                textoNormal
            )
            y += 22

            canvas.drawText(
                "Estado: ${if (tarea.estaCompletada) "Completada" else "Pendiente"}",
                40f,
                y.toFloat(),
                textoNormal
            )
            y += 22

            canvas.drawText(
                "Archivo: ${tarea.archivoNombre ?: "Sin archivo"}",
                40f,
                y.toFloat(),
                textoNormal
            )
            y += 36
        }
    }

    pdfDocument.finishPage(page)

    val archivo = File(context.cacheDir, "tareas.pdf")
    val outputStream = FileOutputStream(archivo)

    pdfDocument.writeTo(outputStream)
    pdfDocument.close()
    outputStream.close()

    compartirArchivo(
        context = context,
        archivo = archivo,
        tipoMime = "application/pdf"
    )
}
